void insertionSort2(int n, int arr_count, int* arr) {
    for (int i = 1; i < n; i++) {
        int valueToInsert = arr[i];
        int j = i - 1;

        while (j >= 0 && arr[j] > valueToInsert) {
            arr[j + 1] = arr[j];
            j--;
        }

        arr[j + 1] = valueToInsert;

        for (int k = 0; k < n; k++) {
            printf("%d", arr[k]);
            if (k < n - 1) printf(" ");
        }
        printf("\n");
    }
}
